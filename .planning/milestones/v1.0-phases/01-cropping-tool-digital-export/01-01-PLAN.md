   ```typescript
   export interface CropState {
     offsetX: number;  // 0-1, fraction of source image at crop frame left edge
     offsetY: number;  // 0-1, fraction of source image at crop frame top edge
     zoomFraction: number; // fraction of source image width that fills crop frame width
   }

   export interface ImageState {
     element: HTMLImageElement | null;
     naturalWidth: number;
     naturalHeight: number;
     url: string;
   }

   export interface ExportTarget {
     widthPx: number;
     heightPx: number;
     dpi: number;
     format: 'png' | 'jpeg';
     quality?: number;
   }

   export interface GuidePosition {
     topFrac: number;    // fraction from top of crop frame
     bottomFrac: number; // fraction from top of crop frame
   }
   ```

2. Create `src/lib/state/spec.ts` with AU_PASSPORT_SPEC constant exactly as in the research doc (lines 299-343). Add derived guide positions:
   ```typescript
   // Guide band positions as fractions from TOP of crop frame
   // Chin zone: chin should be at 80-88% from top (12-20% from bottom)
   // Crown zone: crown should be at 1-17% from top (83-99% from bottom)
   // Derived from: face is 32-36mm of 45mm, centered roughly with chin at bottom portion
   export const chinGuide: GuidePosition = {
     topFrac: 1 - 0.19,   // 0.81 from top
     bottomFrac: 1 - 0.12, // 0.88 from top
   };
   export const crownGuide: GuidePosition = {
     topFrac: 1 - 0.99,   // 0.01 from top
     bottomFrac: 1 - 0.83, // 0.17 from top
   };
   ```
   Note: These guide band positions are derived from the spec ranges. The chin zone spans where the bottom of the chin should fall. The crown zone spans where the top of the skull should fall.
